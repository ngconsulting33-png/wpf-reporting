📋 SQL à exécuter dans Supabase (une seule fois) 
CREATE TABLE semaines (
  id SERIAL PRIMARY KEY,
  cle TEXT UNIQUE NOT NULL,
  sem INTEGER,
  annee INTEGER,
  mois INTEGER,
  mois_nom TEXT,
  donnees JSONB NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
 
ALTER TABLE semaines ENABLE ROW LEVEL SECURITY;
 
CREATE POLICY "Lecture publique" ON semaines
  FOR SELECT USING (true);
 
CREATE POLICY "Ecriture publique" ON semaines
  FOR INSERT WITH CHECK (true);
 
CREATE POLICY "Modification publique" ON semaines
  FOR UPDATE USING (true);
Va dans Supabase → SQL Editor, colle ce code, clique Run.
